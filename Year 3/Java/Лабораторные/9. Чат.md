Создать на основе сокетов клиент/серверное приложение «Чат». Клиент регистрируется на сервере, после чего посылает через сервер сообщения, которые получают все клиенты. Список клиентов хранится на сервере.

Интерфейс у клиентского приложения определить самостоятельно.

```java
@Route("")
public class MainView extends VerticalLayout {
    private Socket socket;
    private BufferedReader in;
    private PrintWriter out;
    private Div messageArea;
    private TextField messageInput;
    private Button sendButton;
    private VerticalLayout loginLayout;
    private VerticalLayout chatLayout;
    private boolean usernameSet = false;

    public MainView() {
        setSizeFull();
        setPadding(true);

        createLoginLayout();
        createChatLayout();

        add(loginLayout);
        chatLayout.setVisible(false);
    }

    private void createLoginLayout() {
        loginLayout = new VerticalLayout();
        loginLayout.setSizeFull();
        loginLayout.setAlignItems(Alignment.CENTER);
        loginLayout.setJustifyContentMode(JustifyContentMode.CENTER);

        TextField usernameField = new TextField("Username");
        usernameField.setPlaceholder("Enter your username");
        usernameField.setWidth("300px");

        Button connectButton = new Button("Connect", e -> {
            String username = usernameField.getValue().trim();
            if (!username.isEmpty()) {
                connectToServer(username);
            } else {
                Notification.show("Please enter a username");
            }
        });
        connectButton.addClickShortcut(Key.ENTER);

        loginLayout.add(usernameField, connectButton);
    }

    private void createChatLayout() {
        chatLayout = new VerticalLayout();
        chatLayout.setSizeFull();
        chatLayout.setPadding(false);

        messageArea = new Div();
        messageArea.setWidthFull();
        messageArea.getStyle()
                .set("flex-grow", "1")
                .set("overflow-y", "auto")
                .set("border", "1px solid #ccc")
                .set("padding", "10px")
                .set("background-color", "#f9f9f9");

        HorizontalLayout inputLayout = new HorizontalLayout();
        inputLayout.setWidthFull();

        messageInput = new TextField();
        messageInput.setPlaceholder("Type your message...");
        messageInput.setWidthFull();

        sendButton = new Button("Send", e -> sendMessage());
        sendButton.addClickShortcut(Key.ENTER);

        inputLayout.add(messageInput, sendButton);
        inputLayout.setFlexGrow(1, messageInput);

        chatLayout.add(messageArea, inputLayout);
        chatLayout.setFlexGrow(1, messageArea);
    }

    private void connectToServer(String username) {
        try {
            socket = new Socket("localhost", 12345);
            in = new BufferedReader(new InputStreamReader(socket.getInputStream()));
            out = new PrintWriter(socket.getOutputStream(), true);

            // Send username as first message
            out.println(username);
            usernameSet = true;

            // Switch to chat view
            remove(loginLayout);
            add(chatLayout);
            chatLayout.setVisible(true);

            // Start listening for server messages
            startMessageListener();

            Notification.show("Connected to server as " + username);

        } catch (IOException e) {
            Notification.show("Failed to connect to server: " + e.getMessage());
            e.printStackTrace();
        }
    }

    private void startMessageListener() {
        Thread listenerThread = new Thread(() -> {
            String serverMessage;
            try {
                while ((serverMessage = in.readLine()) != null) {
                    final String message = serverMessage;
                    getUI().ifPresent(ui -> ui.access(() -> {
                        displayMessage(message);
                        ui.push();
                    }));
                }
            } catch (IOException e) {
                getUI().ifPresent(ui -> ui.access(() -> {
                    displayMessage("Disconnected from server.");
                    messageInput.setEnabled(false);
                    sendButton.setEnabled(false);
                    ui.push();
                }));
            }
        });
        listenerThread.setDaemon(true);
        listenerThread.start();
    }

    private void sendMessage() {
        String message = messageInput.getValue().trim();
        if (!message.isEmpty() && out != null) {
            out.println(message);
            messageInput.clear();
            messageInput.focus();
        }
    }

    private void displayMessage(String message) {
        Div messageDiv = new Div();
        messageDiv.setText(message);
        messageDiv.getStyle()
                .set("margin-bottom", "5px")
                .set("padding", "5px")
                .set("border-radius", "3px")
                .set("background-color", "#fff");
        messageArea.add(messageDiv);
    }

    @Override
    protected void onDetach(com.vaadin.flow.component.DetachEvent detachEvent) {
        super.onDetach(detachEvent);
        closeConnection();
    }

    private void closeConnection() {
        try {
            if (in != null) in.close();
            if (out != null) out.close();
            if (socket != null) socket.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
```

### Server.java
```java

public class Server {
    private static final int PORT = 12345;
    private static final Map<String, ClientHandler> clients = new ConcurrentHashMap<>();

    public static void start() {
        try (ServerSocket serverSocket = new ServerSocket(PORT)) {
            System.out.println("Server started on port " + PORT);

            while (true) {
                Socket socket = serverSocket.accept();
                System.out.println("New client connected: " + socket.getInetAddress());

                // Handle client in a separate thread
                new Thread(new ClientHandler(socket)).start();
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    // Broadcast a message to all connected clients
    public static void broadcast(String message, String senderUsername) {
        for (ClientHandler client : clients.values()) {
//            if (!Objects.equals(client.getUsername(), senderUsername)) {
                client.sendMessage(message);
//            }
        }
    }

    public static void listClients(String sender) {
        ClientHandler client = clients.get(sender);

        StringBuilder builder = new StringBuilder("Clients: ");
        for (ClientHandler clientInList : clients.values()) {
            builder.append(clientInList.getUsername()).append(";");
        }
        client.sendMessage(builder.toString());
    }

    // Add client to map
    public static void registerClient(String username, ClientHandler handler) {
        clients.put(username, handler);
    }

    // Remove client from map
    public static void removeClient(String username) {
        clients.remove(username);
    }
}
```

### ClientHandler.java
```java
public class ClientHandler implements Runnable{
    private Socket socket;
    private String username;
    private PrintWriter out;
    private BufferedReader in;

    public ClientHandler(Socket socket) {
        this.socket = socket;
    }

    public String getUsername() {
        return username;
    }

    @Override
    public void run() {
        try {
            in = new BufferedReader(new InputStreamReader(socket.getInputStream()));
            out = new PrintWriter(socket.getOutputStream(), true);

            out.println("Enter your username: ");
            username = in.readLine();
            out.flush();

            Server.registerClient(username, this);
            System.out.println(username + " joined.");
            Server.broadcast(username + " has joined the chat.", username);

            String message;
            while ((message = in.readLine()) != null) {
                if (message.equals("/clients")) {
                    Server.listClients(username);
                } else {
                    System.out.println(username + ": " + message);
                    Server.broadcast(username + ": " + message, username);

                }
            }

        } catch (IOException e) {
            System.out.println(username + " disconnected.");
        } finally {
            try {
                Server.removeClient(username);
                Server.broadcast(username + " has left the chat.", username);
                socket.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }

    public void sendMessage(String message) {
        out.println(message);
    }
}

```
# Решение
## Сервер
Сервер открывает `ServerSocket`, на который принимаются подключения. На каждое подключение по итогу создаётся поток `ClientHandler`.

Каждый ClientHandler обрабатывает одно подключение (пользователя). Первое сообщение является именем пользователя, которое используется для занесения списка активных клиентов в ConcurrentHashMap. При отключении клиента, он удаляется из ConcurrentHashMap. Каждый клиент может писать в (и читать из) Socket, который передаётся при его создании.
